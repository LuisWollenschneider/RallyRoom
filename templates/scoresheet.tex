% A blank framed box to write a value into: {width}{height}
\newcommand{\writebox}[2]{%
  \setlength{\fboxsep}{0pt}%
  \fcolorbox{rule}{white}{\makebox[#1]{\rule[-2mm]{0pt}{#2}}}}

\newcommand{\chip}[2]{%
  \setlength{\fboxsep}{2pt}%
  \colorbox{#1}{\makebox[6mm][c]{\rule[-1.2mm]{0pt}{4mm}}\color{white}\small\textbf{#2}\makebox[3mm]{}}}

% ── Fillable AcroForm fields ─────────────────────────────────────────────────
% Fields live in normal flow so their widget rectangles land exactly where
% drawn and stay clickable. (borderwidth=0 + a set bordercolor suppresses the
% viewer's default red frame.)
% Bordered numeric entry — looks like the framed score box: {name}{width}{height}
\newcommand{\finput}[3]{%
  \TextField[name=#1,width=#2,height=#3,borderwidth=0.6pt,%
    bordercolor={0.812 0.776 0.729},backgroundcolor={1 1 1},%
    color={0.165 0.125 0.094},charsize=11pt,align=1]{}}
% Borderless text input with an underline rule drawn back under it: {name}{width}
\newcommand{\fline}[2]{%
  \TextField[name=#1,width=#2,height=5mm,borderwidth=0pt,bordercolor={1 1 1},%
    color={0.165 0.125 0.094},charsize=11pt]{}%
  \kern-#2{\color{rule}\rule[-0.8mm]{#2}{0.5pt}}}
% Bordered checkbox: {name}
\newcommand{\fcheckbox}[1]{%
  \CheckBox[name=#1,width=5mm,height=5mm,borderwidth=0.6pt,%
    bordercolor={0.812 0.776 0.729},backgroundcolor={1 1 1}]{}}
